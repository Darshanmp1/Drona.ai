# Drona AI - Core Dependencies
# Requirements for RAG foundation + Content Extraction

# === RAG Core Dependencies ===
# Main dependency for generating text embeddings
# This is a lightweight library that provides pre-trained models
sentence-transformers==2.2.2

# Required for numerical operations and vector math
# The vector store uses this for similarity calculations
numpy==1.24.3

# These come with sentence-transformers but listing for clarity:
# - torch (PyTorch - the deep learning framework)
# - transformers (Hugging Face transformers library)
# - scikit-learn (for some ML utilities)

# === Content Extraction Dependencies ===

# PDF Extraction
# PyPDF2 - Extract text from PDF documents
PyPDF2==3.0.1

# Image OCR (Optical Character Recognition)
# pytesseract - Python wrapper for Tesseract OCR engine
# Note: Tesseract must be installed separately on your system
pytesseract==0.3.10

# Pillow - Image processing library (required for pytesseract)
Pillow==10.0.0

# YouTube Transcript Extraction
# youtube-transcript-api - Fetch video subtitles/transcripts
youtube-transcript-api==0.6.1

# Web Scraping
# beautifulsoup4 - Parse HTML and extract clean text
beautifulsoup4==4.12.2

# requests - Make HTTP requests to fetch web pages
requests==2.31.0

# lxml - Fast HTML/XML parser (used by BeautifulSoup)
lxml==4.9.3

# === Voice Assistant Dependencies ===

# Speech Recognition
# SpeechRecognition - Convert speech to text
# Supports multiple engines (Google, Sphinx, Wit.ai, etc.)
SpeechRecognition==3.10.0

# PyAudio - Required for microphone input with SpeechRecognition
# Note: On Windows, may need to install via: pip install pipwin, then: pipwin install pyaudio
# Or download wheel from: https://www.lfd.uci.edu/~gohlke/pythonlibs/#pyaudio
PyAudio==0.2.13

# Text-to-Speech
# pyttsx3 - Offline text-to-speech conversion
# Works without internet connection, uses native OS speech engines
pyttsx3==2.90
